// Example: Newsletter form placeholder functionality document.addEventListener("DOMContentLoaded", function () { const form = document.querySelector(".newsletter form"); if (form) { form.addEventListener("submit", function (e) { e.preventDefault(); const email = document.getElementById("mail").value.trim(); if (email === "") { alert("Please enter your email address."); } else { alert(`Thank you for subscribing with: ${email}`); form.reset(); } }); } });
